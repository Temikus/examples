1. Create a new rails project and add a task model
   #+NAME: init
     #+BEGIN_SRC sh
     rails new todo
     rails g scaffold task title:string notes:string due:datetime completion:integer
     #+END_SRC

2. Modify the gem file to look like this
   #+NAME: Gemfile
     #+BEGIN_SRC Ruby
     gem 'rails', '4.2.0'
     gem 'sass-rails', '~> 5.0'
     gem 'uglifier', '>= 1.3.0'
     gem 'coffee-rails', '~> 4.1.0'

     gem 'jquery-rails'
     gem 'turbolinks'
     gem 'jbuilder', '~> 2.0'
     gem 'sdoc', '~> 0.4.0', group: :doc

     group :production do
       gem 'pg'
     end

     gem 'capistrano-rails', group: :development

     group :development, :test do
       gem 'byebug'

       gem 'web-console', '~> 2.0'

       gem 'spring'

       gem 'sqlite3'
     end
     #+END_SRC

3. Modify database.yml
   #+NAME: database.yml
     #+BEGIN_SRC yaml
     default: &default
       adapter: sqlite3
       pool: 5
       timeout: 5000

     development:
       <<: *default
       database: db/todo_development.sqlite3

     test:
       database: db/todo_test.sqlite3

     production:
       <<: *default
       adapter: postgresql
       encoding: unicode
       database: todo_production
       username: <%= ENV['PG_ENV_POSTGRES_USER'] %>
       password: <%= ENV['PG_ENV_POSTGRES_PASSWORD'] %>
       host:     <%= ENV['PG_PORT_5432_TCP_ADDR'] %>
     #+END_SRC

4. Create a Dockerfile
   #+NAME: Dockerfile
     #+BEGIN_SRC 
     FROM rails:onbuild

     ENV RAILS_ENV=production

     CMD ["sh", "/usr/src/app/init.sh"]
     #+END_SRC

5. Create an init.sh
    #+Name: init.sh
      #+BEGIN_SRC sh
      export SECRET_KEY_BASE=$(bundle exec rake secret)

      bundle exec rake db:create db:migrate
      bundle exec rails server -b 0.0.0.0
      #+END_SRC

6. Build the docker image
  #+BEGIN_SRC sh
  docker build -t thagomizer/todo .
  #+END_SRC

7. Start up the database container
  #+BEGIN_SRC sh
    docker run --name db -e POSTGRES_PASSWORD=password -e POSTGRES_USER=rails -d postgres
   #+END_SRC

8. Start up the web container
  #+BEGIN_SRC sh
  docker run --name web -d -p 3000:3000 --link db:pg thagomizer/todo
  #+END_SRC

9. Access the site.
  To get the url on linux
  #+BEGIN_SRC sh
  docker ports
  #+END_SRC

  To get the url with bookt2dockr
  #+BEGIN_SRC sh
  boot2docker ip
  #+END_SRC

  Remember the webiste is on port 3000
